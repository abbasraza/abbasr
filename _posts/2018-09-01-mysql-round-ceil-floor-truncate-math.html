floor() --> take down
ceil() --> take up
round() --> either floor() or ceil() whichever is closer 
truncate() --> discard fractional part

SELECT 1.1              as number,
       round(1.1)       as 'round(number)',
       floor(1.1)       as 'floor(number)',
       ceil(1.1)        as 'ceil(number)',
       truncate(1.1, 0) as 'truncate(number, 0)'
union
select 4.6,
       round(4.6),
       floor(4.6),
       ceil(4.6),
       truncate(4.6, 0)
union
select 6.9,
       round(6.9),
       floor(6.9),
       ceil(6.9),
       truncate(6.9, 0)

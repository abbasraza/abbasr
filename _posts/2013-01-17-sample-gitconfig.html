# cat /home/abbas/.gitconfig

[user]
        email = Abbas_Raza@mentor.com
        name = Abbas Raza
[color]
        ui = auto
[alias]
        st = status
        df = diff
        lg = log -p
        lol = log --graph --decorate --pretty=oneline --abbrev-commit
        lola = log --graph --decorate --pretty=oneline --abbrev-commit --all
        br = branch
        ci = commit -s
        co = checkout
        ls = ls-files
        fp = format-patch
        f = fetch
#[format] its a bad thing and will sign-off every "git format-patch" with your name

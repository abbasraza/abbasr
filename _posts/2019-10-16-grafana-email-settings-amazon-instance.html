sudo vim /etc/grafana/grafana.ini

#################################### SMTP / Emailing ##########################
[smtp]
enabled = true
host = email-smtp.eu-west-1.amazonaws.com:587
user = PUT_USER_HERE

# If the password contains # or ; you have to wrap it with trippel quotes. Ex """#password;"""
password = PUT_PASSWORD_HERE
;cert_file =
;key_file =
;skip_verify = false
from_address = grafana@tridonic.com
from_name = Heimdallr Grafana
# EHLO identity in SMTP dialog (defaults to instance_name)
;ehlo_identity = dashboard.example.com




sudo cat /etc/ssmtp/ssmtp.conf
#
# Config file for sSMTP sendmail
#
# The person who gets all mail for userids < 1000
# Make this empty to disable rewriting.
root=postmaster

# The place where the mail goes. The actual machine name is required no
# MX records are consulted. Commonly mailhosts are named mail.domain.com
mailhub=email-smtp.eu-west-1.amazonaws.com:587
rewritedomain=tridonic.com

# Where will the mail seem to come from?
#rewriteDomain=

# The full hostname
hostname=ip-172-31-23-236.eu-west-1.compute.internal
#hostname=www.connecdim.com

# Are users allowed to set their own From: address?
# YES - Allow the user to specify their own From: address
# NO - Use the system generated From: address
AuthUser=HERE_IS_USER_NAME
AuthPass=HERE_IS_PASSWORD

FromLineOverride=YES
UseTLS=YES
UseStartTLS=YES

#Debug=YES
